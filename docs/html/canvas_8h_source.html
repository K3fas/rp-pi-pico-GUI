15// canvas format
16// Note: do not use enum, symbols could not be used by the preprocessor
17#define CANVAS_8 0 // 8-bit pixels
18#define CANVAS_4 1 // 4-bit pixels
19#define CANVAS_2 2 // 2-bit pixels
20#define CANVAS_1 3 // 1-bit pixels
21#define CANVAS_PLANE2 4 // 4 colors on 2 planes
22#define CANVAS_ATTRIB8 5 // 2x4 bit color attributes per 8x8 pixel sample
23// draw functions: bit 0..3 = draw color
24// bit 4 = draw color is background color

26// canvas descriptor

32 int h; // height
33 int wb; // pitch (bytes between lines)
34 u8 format; // canvas format CANVAS_*
35} sCanvas;
