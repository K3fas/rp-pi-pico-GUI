24 template <typename T>
25 concept Stepable = requires(T x) {
26 {
27 ++x
28 } -> std::same_as<T &>;
29 {
30 --x
31 } -> std::same_as<T &>;
32 };

34 template <typename T>
35 concept Constructible = requires {
36 typename std::enable_if<std::is_default_constructible_v<T>>::type;
37 };

40 concept StepableAndConstructible = requires {

143 template <Stepable T>
144 inline void Stepper<T>::prepareButtons()
145 {
146 auto bound = this->GetBounds();
147 auto adjBounds = bound - frameWidth;
148 adjBounds.w = adjBounds.w / 2;
149 _bAdd = new Button(Bounds(adjBounds.x + frameWidth / 2 + adjBounds.w, adjBounds.y, adjBounds.w, adjBounds.h),
150 color,
151 "+",
152 backgroundColor);
153
154 _bSub = new Button(Bounds(adjBounds.x + frameWidth / 2 + adjBounds.w, adjBounds.y, adjBounds.w, adjBounds.h),
155 color,
156 "-",
157 backgroundColor);
158 _bAdd->SetParrent(this);
159 _bSub->SetParrent(this);
160 _bAdd->SetOnClickHandler(MouseEventType::Clicked, add, 1);
161 _bSub->SetOnClickHandler(MouseEventType::Clicked, sub, 1);
162 }
