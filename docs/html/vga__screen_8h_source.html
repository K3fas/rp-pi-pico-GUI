11// video segment (on change update SSEGM_* in define.h)
12typedef struct {
13 u16 width; // SSEGM_WIDTH width of this video segment in pixels (must be multiple of 4, 0=inactive segment)
14 u16 wb; // SSEGM_WB pitch - number of bytes between lines
15 s16 offx; // SSEGM_OFFX display offset at X direction (must be multiple of 4)
16 s16 offy; // SSEGM_OFFY display offset at Y direction
17 u16 wrapx; // SSEGM_WRAPX wrap width in X direction (number of pixels, must be multiply of 4 and > 0)
18 // text modes: wrapx must be multiply of 8
19 u16 wrapy; // SSEGM_WRAPY wrap width in Y direction (number of lines, cannot be 0)
20 const void* data; // SSEGM_DATA pointer to video buffer with image data
21 u8 form; // SSEGM_FORM graphics format GF_*
22 bool dbly; // SSEGM_DBLY double Y (2 scanlines per 1 image line)
23 u16 par3; // SSEGM_PAR3 parameter 3
24 u32 par; // SSEGM_PAR parameter 1
25 u32 par2; // SSEGM_PAR2 parameter 2
26} sSegm;
