55 constexpr uint LED_PINS[] = {6, 7, 8, 9};
56 constexpr uint POTENTIOMETER = 27;
57 constexpr uint POTENTIOMETER_ADC = 1;
58
59 void on_pwm_wrap()
60 {
61 // Clear the interrupt flag that brought us here
62 pwm_clear_irq(pwm_gpio_to_slice_num(LED_PINS[0]));
63 for (size_t i = 0; i < 4; i++)
64 {
65 if (data.led[i])
66 {
67 data.progress[i] = (double)(adc_read() >> 4); // read adc value
68 }
69 }
70 }
71
72 void Init()
73 {
74 // adc initialization
75 adc_init();
76 // make sure GPIO is high-impedance, no pullups etc
77 adc_gpio_init(POTENTIOMETER);
78 // select ADC input
79 adc_select_input(POTENTIOMETER_ADC);
80
81 // Get some sensible defaults for the slice configuration. By default, the
82 // counter is allowed to wrap over its maximum range (0 to 2**16-1)
83 pwm_config config = pwm_get_default_config();
84 // Set divider, reduces counter clock to sysclock/this value
85 pwm_config_set_clkdiv(&config, 4.f);
86
87 for (size_t i = 0; i < 4; i++)
88 {
89 gpio_set_function(LED_PINS[i], GPIO_FUNC_PWM);
90 uint slice_num = pwm_gpio_to_slice_num(LED_PINS[i]);
91 pwm_set_gpio_level(LED_PINS[i], 0);
92 pwm_init(slice_num, &config, true);
93 }
94
95 // Mask our slice's IRQ output into the PWM block's single interrupt line,
96 // and register our interrupt handler
97 pwm_clear_irq(pwm_gpio_to_slice_num(LED_PINS[0]));
98 pwm_set_irq_enabled(pwm_gpio_to_slice_num(LED_PINS[0]), true);
99 irq_set_exclusive_handler(PWM_IRQ_WRAP, on_pwm_wrap);
100 irq_set_enabled(PWM_IRQ_WRAP, true);
101 }
102
103 void Update()
104 {
105 for (size_t i = 0; i < 4; i++)
106 {
107 if (data.state[i])
108 {
109 pwm_set_gpio_level(LED_PINS[i], data.progress[i] * data.progress[i]);
110 }
111 else
112 {
113 pwm_set_gpio_level(LED_PINS[i], 0);
114 }
115 }
116 }
